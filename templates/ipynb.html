def preprocessing(new_data):
    new_data = pd.get_dummies(new_data, columns=['Gender', 'Education','Property_Area','Married','Self_Employed'])
    last_row =new_data.tail(1)
    scaler_new_data = scaler.transform(last_row)
    result = classifier.predict(scaler_new_data)
    return result
